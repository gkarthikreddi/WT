    PHP Cookies

                

    
<?php
// Function to set a cookie with a given name, value, and expiration time
function setCookieValue($name, $value, $expire) {
    setcookie($name, $value, time() + $expire, "/");
}

// Function to get the value of a cookie by its name
function getCookieValue($name) {
    return isset($_COOKIE[$name]) ? $_COOKIE[$name] : null;
}

// Function to clear a cookie by setting its expiration time to the past
function clearCookie($name) {
    setcookie($name, "", time() - 3600, "/");
}

// Check if a form is submitted to set a new cookie
if ($_SERVER["REQUEST_METHOD"] == "POST") {
    // Set a cookie named "user" with the submitted value and a 1-hour expiration time
    $username = $_POST["username"];
    setCookieValue("user", $username, 3600);
}

// Check if a form is submitted to clear the cookie
if (isset($_POST["clear_cookie"])) {
    // Clear the cookie named "user"
    clearCookie("user");
}

// Get the current value of the "user" cookie, if it exists
$userCookieValue = getCookieValue("user");
?>

<?php
    // filename to open
    $fname = "file1.txt";
    // to open file for reading
    $fptr = fopen($fname, "r");
    // to read content of the file
    $content=fread($fptr,filesize($fname));
    // to open for writing
    $fptw = fopen("file2.txt", "w");
    // to write content into another file
    fwrite($fptw,$content);
    echo "<strong>The content copied in file2 is:</strong> <br>". $content;
    //close files
    fclose($fptr);
    fclose($fptw);
?>
